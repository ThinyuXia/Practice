#### 一、背包问题

##### 1.01背包问题

特点：每个物品最多使用一次

![](https://s3.bmp.ovh/imgs/2022/05/18/0e125fb97a47fa35.png)



[原题链接](https://www.acwing.com/problem/content/2/)

##### 朴素版代码

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        final int N = 1010;
        int n = in.nextInt();
        int m = in.nextInt();
        int[] v = new int[N];
        int[] w = new int[N];
        int[][] f = new int[N][N];
        
        //f[0][0 ～ m] = 0,不用手动进行初始化
        
        for(int i = 1;i <= n;i ++){
            v[i] = in.nextInt();
            w[i] = in.nextInt();
        }
        
        for(int i = 1;i <= n;i ++){
            for(int j = 0;j <= m;j ++){
                f[i][j] = f[i - 1][j];
                if(j >= v[i]) f[i][j] = Math.max(f[i][j],f[i - 1][j - v[i]] + w[i]);
            }
        }
        System.out.println(f[n][m]);
    }
}
```



##### 优化版代码

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        final int N = 1010;
        int n = in.nextInt();
        int m = in.nextInt();
        int[] v = new int[N];
        int[] w = new int[N];
        int[] f = new int[N];
        
        //f[0][0 ～ m] = 0,不用手动进行初始化
        
        for(int i = 1;i <= n;i ++){
            v[i] = in.nextInt();
            w[i] = in.nextInt();
        }
        
        //1.删掉第一维的原因是每次计算第i层只会用到i - 1层
        //2.j如果正序计算会导致f[j - v[i]]是第i层的数据，状态被污染
        for(int i = 1;i <= n;i ++)
            for(int j = m;j >= v[i];j --)
                f[j] = Math.max(f[j],f[j - v[i]] + w[i]);
        System.out.println(f[m]);
    }
}
```

##### 3.多重背包问题

![](https://s3.bmp.ovh/imgs/2022/05/19/477dda749e1aa269.png)

[原题连接](https://www.acwing.com/problem/content/4/)

##### 朴素版代码

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int m = in.nextInt();
        int[] v = new int[110];
        int[] w = new int[110];
        int[] s = new int[110];
        
        for(int i = 1;i <= n;i ++){
            v[i] = in.nextInt();
            w[i] = in.nextInt();
            s[i] = in.nextInt();
        }
        
        int[][] f = new int[110][110];
        for(int i = 1;i <= n;i ++)
            for(int j = 0;j <= m;j ++)
                for(int k = 0;k * v[i] <= j && k <= s[i];k ++)
                    f[i][j] = Math.max(f[i][j],f[i - 1][j - k * v[i]] + k * w[i]);
        
        System.out.println(f[n][m]);
        
    }
}
```

##### 时间复杂度：O(n * m * s)

##### 二进制优化版代码

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        final int N  = 24000;
        int n = in.nextInt();
        int m = in.nextInt();
        int[] v = new int[N];
        int[] w = new int[N];
        int[] s = new int[N];
        int[] f = new int[N];
        
        int cnt = 0;
        
        for(int i = 1;i <= n;i ++){
            int a = in.nextInt();
            int b = in.nextInt();
            int c = in.nextInt();
            int k = 1;
            while(k <= c){
                cnt ++;
                v[cnt] = a * k;
                w[cnt] = b * k;
                c -= k;
                k *= 2;
            }
            if(c > 0){
                cnt ++;
                v[cnt] = a * c;
                w[cnt] = b * c;
            }
        }
        
        n = cnt;
        
        for(int i = 1;i <= n;i ++)
            for(int j = m;j >= v[i];j --)
                f[j] = Math.max(f[j],f[j - v[i]] + w[i]);
                
        System.out.println(f[m]);
    }
}
```

##### 时间复杂度：O(n * m * log(s))



##### 4.分组背包

特点：

* 有 N 组物品和一个容量是 V 的背包

* 每组物品有若干个，同一组内的物品最多只能选一个



![](https://s3.bmp.ovh/imgs/2022/05/19/2047945b9199f4e8.png)



```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        final int N = 110;
        int[] s = new int[N];
        int[][] v = new int[N][N];
        int[][] w = new int[N][N];
        int[] f = new int[N];
        int n = in.nextInt();
        int m = in.nextInt();
        for(int i = 1;i <= n;i ++){
            s[i] = in.nextInt();
            for(int j = 1;j <= s[i];j ++){
                v[i][j] = in.nextInt();
                w[i][j] = in.nextInt();
            }
        }
        
        for(int i = 1;i <= n;i ++)
            for(int j = m;j >= 0;j --)
                for(int k = 1;k <= s[i];k ++)
                    if(v[i][k] <= j) f[j] = Math.max(f[j],f[j - v[i][k]] + w[i][k]);
                    
        System.out.println(f[m]);
    }
}
```

```java
        int INF = (int)1e8;
```

##### 2.最长上升子序列

![](https://pic.rmb.bdstatic.com/bjh/3cdfc4f918be9dc95b57be275b37eaa4.png)



[原题链接](https://www.acwing.com/problem/content/897/)

##### 朴素代码

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        final int N = 1010;
        int[] a = new int[N];
        int[] f = new int[N];
        
        int[] g = new int[N]; //记忆数组
        
        for(int i = 1;i <= n;i ++) a[i] = in.nextInt();
        
        for(int i = 1;i <= n;i ++){
            f[i] = 1; //以f[i]结尾的上升子序列长度最小为1
            for(int j = 1;j < i;j ++)
                if(a[i] > a[j]){
                    if(f[i] < f[j] + 1){
                        f[i] = f[j] + 1;
                        g[i] = j;
                    }
                }
        }
        
        int idx = 1;
        for(int i = 1;i <= n;i ++) 
            if(f[idx] < f[i]) idx = i;
        
        
        System.out.println(f[idx]);
        
        //逆序输出最长子序列    
        for(int i = 1,len = f[idx];i <= len;i ++){
            System.out.println(a[idx]);
            idx = g[idx];
        }
        System.out.println();
        
    }
}
```



##### 优化代码

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        final int N = 100010;
        final int INF = (int)1e9;
        Scanner in = new Scanner(System.in);
        int[] a = new int[N];
        int[] q = new int[N];
        int n = in.nextInt();
        for(int i = 0;i < n;i ++) a[i] = in.nextInt();
        int len = 0;
        q[0] = -INF;
        for(int i = 0;i < n;i ++){
            int l = 0,r = len;
            while(l < r){
                int mid = (l + r + 1) >> 1;
                if(q[mid] < a[i]) l = mid;
                else r = mid - 1;
            }
            len = Math.max(len,r + 1);
            q[r + 1] = a[i]; 
        }
        System.out.println(len);
    }
}
```



##### 3.最长公共子序列

![](https://pic.rmb.bdstatic.com/bjh/e3e56fba58cf497c26f93958af5ddc13.png)



[原题链接](https://www.acwing.com/problem/content/899/)

##### 代码

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int m = in.nextInt();
        char[] a = in.next().toCharArray();
        char[] b = in.next().toCharArray();
        int[][] f = new int[1010][1010];
        for(int i = 1;i <= n;i ++){
            for(int j = 1;j <= m;j ++){
                f[i][j] = Math.max(f[i - 1][j],f[i][j - 1]);
                //第三种状态出现的前提是a字符串的第i个字符等于b字符串第j个字符，否则无法构成公共子序列
                if(a[i - 1] == b[j - 1]) f[i][j] = Math.max(f[i][j],f[i - 1][j - 1] + 1);
            }
        }
        
        System.out.println(f[n][m]);
    }
}
```



#### 二、区间DP

##### 1.合并石子

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int[][] f = new int[310][310];
        int[] s = new int[n + 1];
        for(int i = 1;i <= n;i ++) s[i] = s[i - 1] + in.nextInt();
        for(int len = 2;len <= n;len ++){
            for(int start = 1;start + len - 1 <= n;start ++){
                int l = start, r = start + len - 1;
                f[l][r] = (int)1e8;
                for(int k = l;k <= r - 1;k ++){
                    f[l][r] = Math.min(f[l][r],f[l][k] + f[k + 1][r] + s[r] - s[l - 1]);
                }
            }
        }
        
        System.out.println(f[1][n]);
    }
}
```



##### 2.最短编辑距离

![](https://pic.rmb.bdstatic.com/bjh/35fbf3ae205564d3eec0ce638b476073.png)

[原题链接](https://www.acwing.com/problem/content/904/)

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        char[] s1 = (" " + in.next()).toCharArray();
        int m = in.nextInt();
        char[] s2 = (" " + in.next()).toCharArray();
        final int N = 1010;
        int[][] f = new int[N][N];
        
        for(int i = 0;i <= n;i ++) f[i][0] = i;        
        
        for(int i = 0;i <= m;i ++) f[0][i] = i;
        
        for(int i = 1;i <= n;i ++)
            for(int j = 1;j <= m;j ++){
                f[i][j] = Math.min(f[i - 1][j] + 1,f[i][j - 1] + 1);
                if(s1[i] == s2[j]) f[i][j] = Math.min(f[i][j],f[i - 1][j - 1]);
                else f[i][j] = Math.min(f[i][j],f[i - 1][j - 1] + 1);
            }
        System.out.println(f[n][m]);
    }
}
```

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int mod = (int)(1e9 + 7);
        int[] f = new int[1010];
        int n = in.nextInt();
        f[0] = 1;
        for(int i = 1;i <= n;i ++){
            for(int j = i;j <= n;j ++){
                f[j] = (f[j] + f[j - i]) % mod;
            }
        }
        System.out.println(f[n]);
    }
}
```
