##### 1.1单链表(静态)

```java
import java.util.*;
public class Main{
    static int[] val = new int[100010]; //存放结点值
    static int[] ne = new int[100010];  //存放结点的next指针 
    static int head = -1;  //初始化头结点的下一个结点为空结点，用-1表示
    static int idx = 0; //记录当前用到哪个点,每次idx加一相当于new一个新结点
    
    //头插法
    public static void insertToHead(int x){
        val[idx] = x;
        ne[idx] = head;
        head = idx;
        idx ++;
    }
    
    //在下标为k的结点后插入一个节点
    public static void insert(int k,int x){
        ne[idx] = ne[k];
        val[idx] = x;
        ne[k] = idx;
        idx ++;
    }
    
    //删除下标为k的节点的后一个节点
    public static void delete(int k){
        ne[k] = ne[ne[k]];
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int T = in.nextInt();
        while(T -- != 0){
            char opr = in.next().charAt(0);
            if(opr == 'H'){
                int x = in.nextInt();
                insertToHead(x);
            }
            else if(opr == 'I'){
                int k = in.nextInt();
                int x = in.nextInt();
                insert(k - 1,x);
            }else{
                int k = in.nextInt();
                if(k == 0) head = ne[head];
                else delete(k - 1);
            }
        }
        
        for(int i = head;i != -1;i = ne[i]) System.out.print(val[i] + " ");
        
    }
}
```

##### 1.2.双链表(静态)

```java
import java.util.*;
public class Main{
    
    static int[] l = new int[100010];
    static int[] r = new int[100010];
    static int[] val = new int[100010];
    static int idx = 0;

    //在下标为k的结点的右边插入一个结点
    public static void insert(int k,int x){
        val[idx] = x;
        l[idx] = k;
        r[idx] = r[k];
        l[r[k]] = idx;
        r[k] = idx;
        idx ++;
    }
    
    //删除下标为k的结点
    public static void delete(int k){
        l[r[k]] = l[k];
        r[l[k]] = r[k];
    }

    public static void main(String[] args){
        //初始化左端点为0，右端点为1
        r[0] = 1; //0的右边为1
        l[1] = 0; //1的左边为0
        idx  = 2; //由于初始化左右端点，所以idx从2开始
        
        Scanner in = new Scanner(System.in);
        int T = in.nextInt();
        while(T -- != 0){
            String opr = in.next();
            if(opr.equals("L")){
                int x = in.nextInt();
                insert(0,x);
            }else if(opr.equals("R")){
                int x = in.nextInt();
                insert(l[1],x);
            }else if(opr.equals("D")){
                int k = in.nextInt();
                delete(k + 1);
            }else if(opr.equals("IL")){
                int k = in.nextInt();
                int x = in.nextInt();
                insert(l[k + 1],x);
            }else{
                int k = in.nextInt();
                int x = in.nextInt();
                insert(k + 1,x);
            }
        }
        
        for(int i = r[0];i != 1;i = r[i]) System.out.print(val[i] + " ");
        
    }
}
```

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int[] stack = new int[100010];
        int top = 0;
        for(int i = 0;i < n;i ++){
            int x = in.nextInt();
            while(top != 0 && stack[top] >= x) top --;
            if(top != 0) System.out.print(stack[top] + " ");
            else System.out.print("-1 ");
            stack[++ top] = x;
        }
    }
}
```

##### 栈扩展——表达式求值

```java
import java.util.*;
public class Main{
    static Stack<Character> op = new Stack<>();
    static Stack<Integer> num = new Stack<>();
    static Map<Character,Integer> map = new HashMap<>();
    public static void eval(){
        int b = num.pop();
        int a = num.pop();
        char c = op.pop();
        if(c == '+'){
           num.push(a + b);
        }else if(c == '-'){
            num.push(a - b);
        }else if(c == '*'){
            num.push(a * b);
        }else {
            num.push(a / b);
        }
    }
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        String s = in.next();
        map.put('+',1);
        map.put('-',1);
        map.put('*',2);
        map.put('/',2);
        for(int i = 0;i < s.length();i ++){
            char c = s.charAt(i);
            if(Character.isDigit(c)){
                int j = i;
                int n = 0;
                while(j < s.length() && Character.isDigit(s.charAt(j))){
                    n = n * 10 + s.charAt(j) - '0';
                    j ++;
                }
                i = j - 1;
                num.push(n);
            }else if(c == '('){
                op.push(c);
            }else if(c == ')'){
                while(op.peek() != '(') eval();
                op.pop();
            }else{
                while(!op.empty() && op.peek() != '(' && map.get(op.peek()) >= map.get(c)){
                    eval();
                }
                op.push(c);
            }
        }
        while(!op.empty()) eval();
        System.out.println(num.peek());
    }
}
```

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int[] a = new int[1000010];
        int[] q = new int[1000010];
        int n = in.nextInt();
        int k = in.nextInt();
        for(int i = 0;i < n;i ++) a[i] = in.nextInt();
        int head = 0,tail = -1;
        
        StringBuilder bul = new StringBuilder();
        for(int i = 0;i < n;i ++){
            if(head <= tail && i - k + 1 > q[head]) head ++;
            while(head <= tail && a[q[tail]] >= a[i]) tail --;
            q[++ tail] = i;
            
            if(i >= k - 1) bul.append(a[q[head]] + " ");
        }
        
        System.out.println(bul.toString());
        
        head = 0;
        tail = -1;
        bul = new StringBuilder();
        for(int i = 0;i < n;i ++){
            if(head <= tail && i - k + 1 > q[head]) head ++;
            while(head <= tail && a[q[tail]] <= a[i]) tail --;
            q[++ tail] = i;
            
            if(i >= k - 1) bul.append(a[q[head]] + " ");
        }
        System.out.println(bul.toString());
        
    }
}
```

##### 4.kmp算法

[原题链接](https://www.acwing.com/problem/content/833/)

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        char[] p = (" " + in.next()).toCharArray();
        int m = in.nextInt();
        char[] s = (" " + in.next()).toCharArray();
        int[] ne = new int[100010];
        
        //求ne数组过程
        //ne[i]代表以下标i结尾的后缀和以下标1开头的前缀的最大相等长度
        //ne数组中的前缀和后缀不能等于字串本身，所以next[1] = 1;
        for(int i = 2,j = 0;i <= n;i ++){
            while(j != 0 && p[i] != p[j + 1]) j = ne[j];
            if(p[i] == p[j + 1]) j ++;
            ne[i] = j;
        }
        
        StringBuilder ans = new StringBuilder();
        
        //kmp匹配过程
        for(int i = 1,j = 0;i <= m;i ++){
            while(j != 0 && s[i] != p[j + 1]) j = ne[j];
            if(s[i] == p[j + 1]) j ++;
            if(j == n){ //匹配成功
                ans.append(i - n + " ");
                j = ne[j];
            }
        }
        System.out.println(ans.toString());
    }
}
```

##### 7.堆排序

```java
import java.util.*;
public class Main{
    static int[] a = new int[100010];
    static int size = 0;
    public static void down(int n){
        int t = n;
        if(2 * n <= size && a[2 * n] < a[t]) t = 2 * n;
        if(2 * n + 1 <= size && a[2 * n + 1] < a[t]) t = 2 * n + 1;
        if(t != n){
            int t1 = a[n];
            a[n] = a[t];
            a[t] = t1;
            down(t);
        }
    }
    
    public static void up(int n){
        while(n != 1 && a[n / 2] > a[n]){
            int t = a[n / 2];
            a[n / 2] = a[n];
            a[n] = t;
            n /= 2;
        }
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int k = in.nextInt();
        for(int i = 1;i <= n;i ++) a[i] = in.nextInt();
        size = n;
        for(int i = n / 2;i != 0;i --) down(i);
        for(int i = 1;i <= k;i ++){
            System.out.print(a[1] + " ");
            a[1] = a[size --];
            down(1);
        }
    }
}
```

##### 堆排序扩展——模拟堆

```java
 import java.util.*;
 public class Main{
     static int[] a = new int[100010];
     static int[] ph = new int[100010]; //第几次插入的数->堆中下标
     static int[] hp = new int[100010]; //堆中下标->对应第一次插入的树
     static int size = 0;
     static int m = 0;
     
     public static void heapSwap(int n1,int n2){
         int t = ph[hp[n1]];
         ph[hp[n1]] = ph[hp[n2]];
         ph[hp[n2]] = t;
         t = hp[n1];
         hp[n1] = hp[n2];
         hp[n2] = t;
         t = a[n1];
         a[n1] = a[n2];
         a[n2] = t;
     }
     
     public static void up(int n){
         while(n != 1 && a[n / 2] > a[n]){
             heapSwap(n,n / 2);
             n /= 2;
         }
     }
     
     public static void down(int n){
         int t = n;
         if(n * 2 <= size && a[n * 2] < a[t]) t = n * 2;
         if(n * 2 + 1 <= size && a[n * 2 + 1] < a[t]) t = n * 2 + 1;
         if(n != t){
             heapSwap(t,n);
             down(t);
         }
     }
     
     public static void main(String[] args){
         Scanner in = new Scanner(System.in);
         int T = in.nextInt();
         while(T -- != 0){
             String opr = in.next();
             if(opr.equals("I")){
                 int n = in.nextInt();
                 size ++;
                 m ++;
                 a[size] = n;
                 ph[m] = size;
                 hp[size] = m;
                 up(size);
             }else if(opr.equals("PM"))
                System.out.println(a[1]);
             else if(opr.equals("DM")){
                 heapSwap(1,size);
                 size --;
                 down(1);
             }else if(opr.equals("D")){
                 int k = ph[in.nextInt()];
                 heapSwap(k,size);
                 size --;
                 up(k);
                 down(k);
             }else{
                 int k = ph[in.nextInt()];
                 int x = in.nextInt();
                 a[k] = x;
                 up(k);
                 down(k);
             }
         }
     }
 }
```
