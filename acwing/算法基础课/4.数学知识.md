#### 一、质数

##### 1.试除法判定质数

```java
public static boolean check(int n){
  if(n < 2) return false;
  for(int i = 2;i <= n / i;i ++){
    if(n % i == 0) return false;
  }
  return true;
}
```

##### 时间复杂度：O(sqrt(n))  

##### 空间复杂度： O(1)



##### 2.拭除法求质因数

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int T = in.nextInt();
        while(T -- != 0){
            int n = in.nextInt();
            for(int i = 2;i <= n / i;i ++)
              //1.能整除n的i一定是质数
              //理由：枚举到i时，n中2到i - 1之间的质因子已经被删除完毕，因为i是n的因子，但是n中不包含2到i - 1之间的质因子，所以i一定是质数
                if(n % i == 0){
                    int s = 0;
                    while(n % i == 0){
                        n /= i;
                        s ++;
                    }
                    System.out.println(i + " " + s);
                }
            if(n != 1) System.out.println(n + " " + 1);    
            System.out.println();
        }
    }
}
```

##### 2.1埃氏筛法

```java
// 思路证明：为什么删除完的数组中所有数均为质数，反证法证明：若删除完的数组中
// p不是质数，则它一定是2 ～ (p - 1) 中某一个质数的若干倍，但是2 ～ (p - 1)中的
// 的每一个质数的倍数已经删完了，所以p一定是质数，证毕。
import java.util.*;
public class Main{
    
    static int[] p = new int[1000010];
    static boolean[] s = new boolean[1000010];
    static int cnt = 0;
    static void get_primes(int n){
        for(int i = 2;i <= n;i++){
            if(!s[i]) {
                p[cnt++] = i;
                for(int j = i;j <= n;j += i) s[j] = true;
            }
        }
        System.out.print(cnt);
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        get_primes(n);
    }
}
```



##### 2.2线性筛法

```java
//核心思想：每个合数都被他的最小质因子筛掉
import java.util.*;
public class Main{
    static int[] primes = new int[1000010];
    static boolean[] s = new boolean[1000010];
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int cnt = 0;
        for(int i = 2;i <= n;i ++){
            if(! s[i]) primes[cnt ++] = i;
            for(int j = 0;i * primes[j] <= n;j ++){
                s[i * primes[j]] = true;
                if(i % primes[j] == 0) break;
            }
        }
        System.out.println(cnt);
    }
}
```

##### 2.约数个数

```java
//大于1的正整数可以背分解成不同质数的指定幂次相乘的形式：p1^a1 * p2^a2 ... pn^an
//每个质数取不同幂次时都是一个约数
//约数个数为(a1 + 1)*(a2 + 1) ... *(an + 1)
//乘法原理 
import java.util.*;
public class Main{
    static int mod = 1000000000 + 7;
    static HashMap<Integer,Integer> primes = new HashMap<>();
    public static void getPrimes(int n){
        for(int i = 2;i <= n / i;i ++){
            while(n % i == 0){
                primes.put(i,primes.getOrDefault(i,0) + 1);
                n /= i;
            }
        }
        if(n != 1){ //对于质数进行特判
            System.out.println(n);
            primes.put(n,primes.getOrDefault(n,0) + 1);
        }
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int T = in.nextInt();
        while(T -- != 0){
            int n = in.nextInt();
            getPrimes(n);
        }
        long ans = 1;
        for(int n : primes.values())
            ans = (ans * (n + 1)) % mod;
        System.out.println(ans);
    }
}
```



##### 3.约数之和

```java
//加法原理
//正整数n的约数之和可以表示为：(p1^0 + p1^1 + p1^2 + ... + p1^a1) * (p2^0 + p2^1 + p2^2 + ... + p2^a1) * (pn^0 + pn^1 + pn^2 + ... + pn^an)
import java.util.*;
public class Main{
    static HashMap<Integer,Integer> primes = new HashMap<>();
    static int mod = (int)(1e9 + 7);
    
    public static void getDivisors(int n){
        for(int i = 2;i <= n / i;i ++){
            while(n % i == 0){
                primes.put(i,primes.getOrDefault(i,0) + 1);
                n /= i;
            }
        }
        if(n != 1)
            primes.put(n,primes.getOrDefault(n,0) + 1);
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int T = in.nextInt();
        while(T -- != 0){
            int n = in.nextInt();
            getDivisors(n);
        }
        long ans = 1;
        for(int key : primes.keySet()){
            int n = primes.get(key);
            long t = 1;
            while(n -- != 0){
                t = (t * key + 1) % mod;
            }
            ans = (ans * t) % mod;
        }
        
        System.out.println(ans);
    }
    
}
```



#### 三、欧拉函数

##### 1.欧拉函数

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int T = in.nextInt();
        while(T -- != 0){
            int n = in.nextInt();
            int ans = n;
            for(int i = 2;i <= n / i;i ++){
                if(n % i == 0){
                    ans = ans / i * (i - 1);
                    while(n % i == 0){
                        n /= i;
                    }
                }
            }
            if(n != 1) ans = ans / n * (n - 1);
            System.out.println(ans);
        }
        
    }
}
```



##### 2.筛法求欧拉函数

```java
import java.util.*;
public class Main{
    static int[] phi = new int[1000010];
    static boolean[] s = new boolean[1000010];
    static int[] p = new int[1000010];
    static int cnt = 0;
    
    public static long get_eulers(int n){
        long ans = 0;
        phi[1] = 1; //特殊处理
        for(int i = 2;i <= n;i ++){
            if(! s[i]){
                if(! s[i]) p[cnt ++] = i;
                phi[i] = i - 1;
            }
            for(int j = 0;i * p[j] <= n;j ++){
                s[i * p[j]] = true;
                if(i % p[j] == 0) 
                    {   
                        phi[p[j] * i] = phi[i] * p[j];
                        break;
                    }
                else{
                    phi[p[j] * i] = phi[i] * (p[j] - 1); 
                }
            }
        }
        for(int i = 1;i <= n;i ++ ) ans += phi[i];
        return ans;
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        System.out.println(get_eulers(n));
        
    }
}
```



#### 四、快速幂

```java
import java.util.*;
public class Main{
    public static long qmi(long a,int k,int p){
        long ans = 1;
        while(k != 0){
            if((k & 1) == 1) ans = ans * a % p;
            k >>= 1;
            a = a * a % p;
        }
        return ans;
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int T = in.nextInt();
        while(T -- != 0){
            int a = in.nextInt();
            int k = in.nextInt();
            int p = in.nextInt();
            System.out.println(qmi(a,k,p));
        }
    }
}
```
