```java
public class Main{
    static final int N = 10;
    static int[] path = new int[N];
    static boolean[] s = new boolean[N];
    static int n;
    public static void dfs(int u){

        if(u == n){
            for(int i = 0;i < n;i++)
                System.out.print(path[i] + " ");
            System.out.println();
        }

        for(int i = 1;i <= n;i ++){
            if(! s[i]){
                path[u] = i;
                s[i] = true;
                dfs(u + 1);
                s[i] = false;
            }
        }
    }

    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        n = in.nextInt();
        dfs(0);

    }
}
```

##### 1.2八皇后问题

做法1:与全排列思路类似，先枚举每行中皇后的位置，再根据是否可以被攻击到进行剪枝

```java
import java.util.*;
public class Main{
    static char[][] grid = new char[10][10];
    static boolean[] col = new boolean[10];
    static boolean[] dg = new boolean[20];
    static boolean[] udg = new boolean[20];
    static int n;
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        n = in.nextInt();
        for(int i = 0;i < n;i ++) Arrays.fill(grid[i],'.');
        dfs(0);
    }
    
    public static void dfs(int u){
        if(u == n){
            for(int i = 0;i < n;i ++){
                for(int j = 0;j < n;j ++) System.out.print(grid[i][j]);
                System.out.println();
            }
            System.out.println();
            return;
        }
        for(int i = 0;i < n;i ++){
            if(! col[i] && ! dg[u + i] && ! udg[u - i + n]){
                grid[u][i] = 'Q';
                col[i] = dg[u + i] = udg[u - i + n] = true;
                dfs(u + 1);
                col[i] = dg[u + i] = udg[u - i + n] = false;
                grid[u][i] = '.';
            }
        }
    }
}

```

 ```java
 public class Main{
     static int n;
     static boolean[] col = new boolean[10];
     static boolean[] row = new boolean[10];
     static boolean[] dg = new boolean[20];
     static boolean[] udg = new boolean[20];
     static char[][] g = new char[20][20];
     public static void main(String[] args){
         Scanner in = new Scanner(System.in);
         n = in.nextInt();
         dfs(0,0,0);
     }
     public static void dfs(int x,int y,int cnt){
         if(cnt > n) return;
         if (y == n) {
             y = 0; 
             x ++;
         }
         if(x == n){
             if(cnt == n){
                 for(int i = 0;i < n;i ++){
                     for(int j = 0;j < n;j ++)
                         System.out.print(g[i][j]);
                     System.out.println();
                 }
                 
                 System.out.println();
             }
             return;
         }
         
         g[x][y] = '.';
         dfs(x,y + 1,cnt); //不放皇后
         if (!row[x] && !col[y] && !dg[y - x + n] && !udg[x + y])
             {
                 row[x] = col[y] = dg[y - x + n] = udg[x + y] = true;
                 g[x][y] = 'Q';
                 dfs(x, y + 1, cnt + 1); //放皇后
                 g[x][y] = '.';
                 row[x] = col[y] = dg[y - x + n] = udg[x + y] = false;
             }
     }
 }
 ```

##### 1.3 树与图的深度优先遍历

[原题链接](https://www.acwing.com/problem/content/848/)

```java
import java.util.*;
public class Main{
    static int N = 100010;
    static int[] h = new int[N]; //采用邻接表的方式存储图，h数组中的节点为链表的头节点
    static int[] e = new int[N * 2]; //存储该点的值，用于找到该点的头结点指针
    static int[] next = new int[N * 2]; //存储h链表中头节点的后续节点
    static boolean[] vis = new boolean[N]; //标记当前节点是否访问过
    static int idx; //节点编号
    static int n;
    
    static int ans = Integer.MAX_VALUE;
    
    public static void add(int a,int b){
        e[idx] = b; //生成一个新结点，该节点的值为b
        next[idx] = h[a]; //该节点的next指针指向头节点指向的节点
        h[a] = idx; //头节点指向idx节点
        idx ++; //节点编号自增
    }
    
    //返回以节点u为根结点的树的节点个数
    public static int dfs(int u){
        vis[u] = true;
        int sum = 1;
        int size = 0;
        for(int i = h[u];i != -1;i = next[i]){
            int j = e[i];
            if(! vis[j]){
                int s = dfs(j);
                size = Math.max(s,size);
                sum += s;
            }
        }
        
        size = Math.max(size,n - sum);
        ans = Math.min(size,ans);
        return sum;
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        Arrays.fill(h,-1);
        n = in.nextInt();
        for(int i = 0;i < n - 1;i ++){
            int a = in.nextInt();
            int b = in.nextInt();
            add(a,b);
            add(b,a);
        }
        
        dfs(1);
        System.out.println(ans);
    }
}
```

##### 2.1迷宫问题

```java
import java.util.*;
public class Main{
    static final int N = 110;
    static int[][] grid = new int[N][N]; //存储地图
    static int[][] d = new int[N][N]; //存储各点到起点的距离
    static Pair[] q = new Pair[N * N]; //维护一个队列
    static Pair[][] prev = new Pair[N][N]; //存储路径的数组，每个元素存的是路径中走向自己的元素
    static int m;
    static int n;
    
    public static int bfs(){
        int[] dx = {-1,0,1,0}; //方向数组
        int[] dy = {0,1,0,-1};
        int hh = 0,tt = 0; //队头和队尾指针
        d[0][0] = 0; //初始化起点的距离为0
        q[0] = new Pair(0,0); //将起点入队
        while(hh <= tt){
            Pair t = q[hh ++]; //第一个元素出队
            for(int i = 0;i < 4;i ++){
                int x = t.x + dx[i]; //判断向四个方向是否可以走
                int y = t.y + dy[i];
                if(x >= 0 && x < m && y >= 0 && y < n && d[x][y] == -1 && grid[x][y] == 0){
                    d[x][y] = d[t.x][t.y] + 1;
                    q[++ tt] = new Pair(x,y); //将当前点入队，要使用++ tt,否则会覆盖元素
                    prev[x][y] = t;
                }
            }
        }
        return d[m - 1][n - 1];
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        m = in.nextInt();
        n = in.nextInt();
        
        for(int i = 0;i < m;i ++)
            for(int j = 0;j < n;j ++) grid[i][j] = in.nextInt();
        
        for(int i = 0;i < m;i ++) Arrays.fill(d[i],-1);
        System.out.println(bfs());
        
        //输出路径
        int x = m - 1;
        int y = n - 1;
        while(x != 0 || y != 0){
            System.out.println(x + " " + y);
            Pair t = prev[x][y];
            x = t.x;
            y = t.y;
        }
    }
}

class Pair{
    int x;
    int y;
    public Pair(int x,int y){
        this.x = x;
        this.y = y;
    }
}
```

```java
import java.util.*;
public class Main{
    static final int N = 100010;
    static int[] h = new int[N];
    static int[] e = new int[2 * N];
    static int[] next = new int[2 * N];
    static int[] d = new int[N]; //节点值表示该节点到1号节点的距离
    static int[] q = new int[N]; //bfs维护的队列
    static int idx;
    static int n;
    static int m;
    
    public static void add(int a,int b){
        e[idx] = b;
        next[idx] = h[a];
        h[a] = idx;
        idx ++;
    }
    
    public static void bfs(){
        int hh = 0,tt = 0;
        Arrays.fill(d,-1);
        q[0] = 1; //将第一个节点入队
        d[1] = 0;
        while(hh <= tt){
            int t = q[hh ++];
            for(int i = h[t];i != -1;i = next[i]){
                int j = e[i];
                if(d[j] == -1){
                    d[j] = d[t] + 1;
                    q[++ tt] = j;
                }
            }
        }
        
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        Arrays.fill(h,-1);
        n = in.nextInt();
        m = in.nextInt();
        while(m -- != 0){
            int a = in.nextInt();
            int b = in.nextInt();
            add(a,b);
        }
        
        bfs();
        
        System.out.println(d[n]);
    }
    
}
```

##### 2.3有向图的拓扑序列——bfs扩展

```java
import java.util.*;
public class Main{
    static int N = 100010;
    static int[] h = new int[N];
    static int[] e = new int[N * 2];
    static int[] next = new int[N * 2];
    static boolean[] vis = new boolean[N];
    static int[] q = new int[N];
    static int[] d = new int[N]; //存储节点的入度
    static int ans = Integer.MAX_VALUE;
    static int idx;
    static int n;
    static int m;
    
    public static void add(int a,int b){
        e[idx] = b; //生成一个新节点
        next[idx] = h[a]; 
        h[a] = idx;
        idx ++;
    }

    public static boolean topsort(){
        int hh = 0,tt = 0;
        
        for(int i = 1;i <= n;i ++){
            if(d[i] == 0) q[tt ++] = i; //度为0的起点放队列前面
        }
        
        while(hh <= tt){
            int t = q[hh ++];
            for(int i = h[t];i != -1;i = next[i]){
                int j = e[i];
                d[j] --;
                if(d[j] == 0) q[tt ++] = j;
            }
        }
        
        return tt == n;
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        
        Arrays.fill(h,-1);
        
        n = in.nextInt();
        m = in.nextInt();
        for(int i = 0;i < m;i ++){
            int a = in.nextInt();
            int b = in.nextInt();
            add(a,b);
            d[b] ++; //b节点的入度加1
        }
        if(topsort()){
            for(int i = 0;i < n;i ++) System.out.print(q[i] + " ");
            System.out.println();
        }else{
            System.out.println(-1);
        }
    }
}
```

##### 2.4八数码

```java
import java.util.*;
public class Main{
    static final int N = 200010;
    static int[] dx = {-1,0,1,0};
    static int[] dy = {0,1,0,-1};
    static String[] q = new String[N]; //初始化队列
    
    public static int bfs(String start){
        String end = "12345678x"; //目标状态
        int hh = 0,tt = 0; //初始化队头和队尾
        Map<String,Integer> dist = new HashMap<>(); //存储任意状态到起始状态的距离
        q[0] = start; //起始状态入队
        dist.put(start,0); //初始化起始状态到它自身的距离为0
        while(hh <= tt){
            String t = q[hh ++]; 
            
            int distance = dist.getOrDefault(t,0);
            
            if(t.equals(end)) return distance; //搜索到目标状态，返回结果
            
            char[] tArr = t.toCharArray();
            int k = t.indexOf('x'); //找到t状态中x的下标
            int x = k / 3, y = k % 3; //将一维下标转换成二维
            for(int i = 0;i < 4;i ++){ //开始状态转换
                int nx = x + dx[i];
                int ny = y + dy[i];
                if(nx >= 0 && nx < 3 && ny >= 0 && ny < 3){
                    char c = tArr[nx * 3 + ny];
                    tArr[nx * 3 + ny] = tArr[k];
                    tArr[k] = c;
                    t = new String(tArr);
                    if(dist.getOrDefault(t,0) == 0){
                        dist.put(t,distance + 1);
                        q[++ tt] = t;
                    }
                    c = tArr[nx * 3 + ny];
                    tArr[nx * 3 + ny] = tArr[k];
                    tArr[k] = c;
                    t = new String(tArr);
                }
            }
        }
        return -1;
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        String s = "";
        for(int i = 0;i < 9;i ++) s += in.next();
        System.out.println(bfs(s));
    }
}
```

##### 3.1.1dijkstra朴素法求最短路

```java
import java.util.*;
public class Main{
    static final int N = 510;
    static final int INF = (int)2e8; //定义正无穷，如果定义成Integer.MAX_VALUE会溢出
    static int[][] g = new int[N][N];
    static boolean[] s = new boolean[N];
    static int[] d = new int[N];
    static int n;
    
    //dijkstra算法分三步
    //1.初始化距离：各点到起点的距离为正无穷
    //2.找到所有未确定最短距离的点中距离起点距离最短点
    //3.根据第二步找到的点更新其他所有点到起点的最短距离
    
    public static int dijkstra(){
        Arrays.fill(d,INF);
        d[1] = 0;
        for(int i = 0;i < n;i ++){
            int t = -1;
            for(int j = 1;j <= n;j ++){
                if(! s[j] && (t == -1 || d[t] > d[j])) 
                    t = j;
            }
            
            s[t] = true;
            
            for(int j = 1;j <= n;j ++){
                    d[j] = Math.min(d[j],d[t] + g[t][j]);
            }
            
        }
        if(d[n] == INF) return -1;
        return d[n];
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        n = in.nextInt();
        int m = in.nextInt();
        for(int i = 1;i <= n;i ++) Arrays.fill(g[i],INF);
        while(m -- != 0){
            int a = in.nextInt();
            int b = in.nextInt();
            int c = in.nextInt();
            g[a][b] = Math.min(g[a][b],c);
        }
        int ans = dijkstra();
        System.out.println(ans);
    }
}
```
